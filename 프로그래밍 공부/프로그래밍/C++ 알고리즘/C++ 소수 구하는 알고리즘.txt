#include <iostream>
#include <algorithm>
#include <vector>
#include <string.h>
#include <string>
#include <queue>
using namespace std;


int main()
{
	vector<int> v;
	int n = 0, con = 0, cnt = 0;
	bool Sosu = true;

	for (int i = 0; i < n; i++)
	{
		if (v[i] == 1)
		{
			continue;
		}

		for (int j = 2; j < v[i] + 1; j++)
		{
			con = v[i] % j;
			if (v[i] == j)
			{
				break;
			}
			if (con == 0)
			{
				Sosu = false;
			}
		}

		if (Sosu)
		{
			cnt++;
		}
		Sosu = true;
	}
	
	return 0;
}
